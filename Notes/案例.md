```js
    <script>
        /*  1. 输入框的聚焦(focus)、失焦(blur) 事件设置边框的颜色
         2. 输入框输入的时候要判断 “发布” 按钮是否可点击
         3. 回车进行发布操作
         4. 点击"发布"按钮，进行发布操作
         5. 点击删除按钮，删除当前这条数据 */

        //  3.加载进来判断 按钮是否可用
        checkBtnFn();
        var flag = false;
        // 1.聚焦
        $('textarea').focus(function () {
            $(this).addClass('active');
        })
        // 2.失焦
        $('textarea').blur(function () {
            $(this).removeClass('active');
        })

        // 4. 回车进行发布操作
        $('textarea').on('keyup', function (e) {
            checkBtnFn();
            // 按enter发布
            // console.log(e.keyCode);
            if (e.keyCode == 13) {
                releaseFn();
            }
        })
        // 6.使用点击发布进行发布
        $('div').on('click', function () {
            if (flag) {
                releaseFn();
            }
        })

        // 5.点击删除按钮，删除当前这条数据 
        $('ul').on('click', 'button', function () {
            // console.log($('button'));
            // 移除当前button的父亲
            $(this).parent().remove();
        })

        // 检查按钮是否可用
        function checkBtnFn() {
            // 获得文本域内容
            var text = $.trim($('textarea').val());
            // console.log(text);
            if (text) {
                flag = true;
                $('div').addClass('active');
                // $('ul').before('<li><p><p><li>')
            } else {
                flag = false;
                $('div').removeClass('active');
            }
        }

        // 发布操作
        function releaseFn() {
            var text = $.trim($('textarea').val());
            if (text) {
                // console.log($('ul'));
                // 将li放到ul
                $('ul').prepend(`<li><p>${text}</p><button>X</button></li>`);
                // 删除类名让按钮不可用
                $('div').removeClass('active');
            } else {
                alert('请重新输入');
            }
            // 清空文本
            $('textarea').val('');
        }
    </script>
```

```html
            // 要定义到全局那里 这样 offsetTop值才不会变0
            var boxTop = box.offsetTop;
            // 滚动事件 onscroll
            window.onscroll = function () {

                var winTop = document.documentElement.scrollTop;

                if (winTop >= boxTop) {
                    box.style.position = 'fixed';
                    box.style.top = '0';
                } else {
                    box.style.position = 'static';
                }
            }
        }
```
